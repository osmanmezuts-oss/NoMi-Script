                            const data = localStorage.getItem(key);
                            if (data) espacioLiberado += data.length;
                            localStorage.removeItem(key);
                            eliminados++;
                            const chatsKey = 'nomi_chats_list_' + partes.slice(1, 3).join('_');
                            const chats = JSON.parse(localStorage.getItem(chatsKey) || '[]');
                            const nuevaLista = chats.filter(f => f !== fechaStr);
                            if (nuevaLista.length !== chats.length) {
                                localStorage.setItem(chatsKey, JSON.stringify(nuevaLista));
                            }
                        }
                    } catch (e) {}
                }
            }
        });
        if (eliminados > 0) {
            console.log(`🧹 Limpieza automática: ${eliminados} historiales eliminados. Espacio liberado: ~${Math.round(espacioLiberado/1024)} KB`);
        }
        return { eliminados, espacioLiberado };

    // ======== OBTENER UBICACIÓN POR GPS ========
    function obtenerUbicacionPorGPS() {
        return new Promise((resolve, reject) => {
            if (!navigator.geolocation) {
                reject(new Error('Geolocalización no soportada por este navegador.'));
                return;
            }
            navigator.geolocation.getCurrentPosition(
                (pos) => {
                    const lat = pos.coords.latitude;
                    const lon = pos.coords.longitude;
                    fuenteUbicacion = 'gps';
                    console.log(`📍 Ubicación por GPS: lat ${lat}, lon ${lon}`);
                    fetch(`https://api.bigdatacloud.net/data/reverse-geocode-client?latitude=${lat}&longitude=${lon}&localityLanguage=es`)
                        .then(res => res.json())
                        .then(data => {
                            const ciudad = data.city || data.locality || data.principalSubdivision || 'Ubicación desconocida';
                            const pais = data.countryName || '';
                            resolve({
                                lat: lat,
                                lon: lon,
                                ciudad: ciudad,
                                pais: pais,
                                timestamp: Date.now(),
                                fuente: 'gps'
                            });
                        })
                        .catch(() => {
                            resolve({
                                lat: lat,
                                lon: lon,
                                ciudad: 'Ubicación desconocida',
                                pais: '',
                                timestamp: Date.now(),
                                fuente: 'gps'
                            });
                        });
                },
                (error) => {
                    if (error.code === 1) {
                        reject(new Error('Permiso denegado. Activa la ubicación en los ajustes del navegador.'));
                    } else if (error.code === 3) {
                        reject(new Error('Timeout. Reintentando...'));
                    } else {
                        reject(new Error('Error al obtener ubicación: ' + error.message));
                    }
                },
                { enableHighAccuracy: true, timeout: 15000, maximumAge: 60000 }
            );
        });
    }

    async function buscarWeb(consulta) {

    async function procesarBusqueda(consulta) {
        if (apiKeyActual !== API_KEY_INTEGRADA) {
            agregarMensaje('bot', '❌ La búsqueda web solo está disponible con la clave integrada de la app.');
            return;

            let ubicacionCoordenadas = '';
            if (ubicacionActivada && ubicacionActual) {
                ubicacionTexto = `${ubicacionActual.ciudad}, ${ubicacionActual.pais}`;
                ubicacionCoordenadas = `${ubicacionActual.lat}, ${ubicacionActual.lon}`;
            }

            // 🔥 CONSTRUIR CONSULTA CON UBICACIÓN EXPLÍCITA
            let consultaFinal = consulta;
            if (ubicacionCoordenadas && requiereUbicacion(consulta)) {
                consultaFinal = `${consulta} en ${ubicacionTexto} (coordenadas: ${ubicacionCoordenadas})`;
            }

            const resultados = await buscarWeb(consultaFinal);
            if (!resultados || !resultados.results || resultados.results.length === 0) {
                const msg = `❌ No se encontraron resultados para esa consulta.`;
                historial.push({role: 'assistant', content: msg});
                guardarHistorial(historial);
                agregarMensaje('bot', msg);
                return;
            }
            const resultadosTexto = resultados.results.map((r, i) => {
                return `${i+1}. ${r.title || 'Sin título'}\n   ${r.content || 'Sin descripción'}`;
            }).join('\n\n');
            
            // 🔥 PROMPT CON INSTRUCCIÓN EXPLÍCITA DE UBICACIÓN GPS
            const prompt = `El usuario se encuentra en ${ubicacionTexto} (coordenadas GPS: ${ubicacionCoordenadas}). **DEBES usar ESTA ubicación para todas las consultas de clima y eventos locales.** Ignora cualquier otra ubicación que puedas inferir de la IP.\n\nInvestigué en la web sobre: "${consultaFinal}". Estos son los resultados obtenidos:\n\n${resultadosTexto}\n\nPor favor, ofrezca una respuesta clara, concisa y en un tono profesional pero cercano. Evite el uso excesivo de tablas o datos innecesarios. Resume la información más importante en 2-3 párrafos. Si hay datos numéricos (temperatura, precios, etc.), menciónelos de forma fluida dentro de la conversación. Mantenga un tono de colaboración entre iguales, sin tuteo excesivo.`;

            const respuestaIA = await llamarIA(prompt);
            const respuestaConIndicador = `🔍 ${respuestaIA}`;
            historial.push({role: 'assistant', content: respuestaConIndicador});
            guardarHistorial(historial);
            agregarMensaje('bot', respuestaConIndicador);
        } catch (error) {
            const msg = `❌ Error en la búsqueda: ${error.message}`;
            historial.push({role: 'assistant', content: msg});
            guardarHistorial(historial);
            agregarMensaje('bot', msg);
        }
    }

    // ======== FUNCIÓN PARA LLAMAR A LA IA ========
    async function llamarIA(mensaje) {
        try {
            const respuesta = await fetch(urlBaseActual + '/chat/completions', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': 'Bearer ' + apiKeyActual
                },
                body: JSON.stringify({
                    model: modeloActual,
                    messages: [{role: 'user', content: mensaje}],
                    stream: false,
                    max_tokens: 1024
                })
            });
            if (!respuesta.ok) {
                const errorText = await respuesta.text();
                throw new Error(`Error ${respuesta.status}: ${errorText}`);
            }
            const datos = await respuesta.json();
            if (datos.choices && datos.choices[0]) {
                return datos.choices[0].message.content;
            } else {
                throw new Error(datos.error?.message || 'Error en la IA');
            }
        } catch (error) {
            registrarError('api', error.message, `Modelo: ${modeloActual}, URL: ${urlBaseActual}`);
            throw error;
        }
    }

    // ======== CONFIGURAR TECLADO ========
    function configurarTeclado() {
        if (!window.visualViewport) return;
        window.visualViewport.addEventListener('resize', () => {
            const win = document.getElementById('nomi-chat');
            if (!win || win.style.display !== 'flex') return;
            const viewportHeight = window.visualViewport.height;
            const keyboardHeight = window.innerHeight - viewportHeight;
            if (keyboardHeight > 100) {
                const nuevoTop = Math.max(10, window.innerHeight - keyboardHeight - win.offsetHeight - 20);
                win.style.top = nuevoTop + 'px';
            } else {
                if (posicionOriginalVentana !== null) {
                    const maxTop = window.innerHeight - win.offsetHeight - 20;
                    const topRestaurado = Math.min(posicionOriginalVentana, maxTop);
                    win.style.top = Math.max(10, topRestaurado) + 'px';
                }
            }
        });
    }

    // ======== ACTUALIZAR UBICACIÓN ========
    async function actualizarUbicacion(silencioso = false) {
        if (!ubicacionActivada) {
            if (!silencioso) {
                mostrarNotificacionTemporal('📍 La ubicación está desactivada. Actívala en el menú (⚙️).');
            }
            return null;
        }
        try {
            const ubicacion = await obtenerUbicacionPorGPS();
            fuenteUbicacion = ubicacion.fuente || 'gps';
            setUbicacion(ubicacion);
            if (!silencioso) {
                mostrarNotificacionTemporal(`📍 Ubicación actualizada: ${ubicacion.ciudad}, ${ubicacion.pais} (${fuenteUbicacion === 'gps' ? 'GPS' : 'IP'})`);
                // 🔥 Mensaje de confirmación en el chat
                agregarMensaje('bot', `📍 Ubicación establecida: **${ubicacion.ciudad}, ${ubicacion.pais}** (${fuenteUbicacion === 'gps' ? 'GPS' : 'IP'}). Las consultas de clima usarán esta ubicación.`);
            }
            actualizarBarraUbicacion();
            inyectarContextoUbicacion(ubicacion);
            return ubicacion;
        } catch (error) {
            const mensajeError = error.message.includes('Permiso denegado') ? '📍 Permiso denegado. Actívalo en ajustes.' : `📍 Error: ${error.message}`;
            actualizarBarraUbicacionConError(mensajeError);
            if (!silencioso) {
                mostrarNotificacionTemporal(`❌ ${error.message}`);
            }
            registrarError('gps', error.message, 'Ubicación');
            return null;
        }
    }

    // ======== INYECTAR CONTEXTO DE UBICACIÓN EN EL HISTORIAL ========
    function inyectarContextoUbicacion(ubicacion) {
        if (!ubicacion) return;
        const fuente = ubicacion.fuente === 'gps' ? 'GPS' : 'IP aproximada';
        // 🔥 INSTRUCCIÓN EXPLÍCITA PARA LA IA
        const mensajeContexto = `📍 Ubicación del usuario: ${ubicacion.ciudad}, ${ubicacion.pais} (${fuente}). Coordenadas GPS: ${ubicacion.lat}, ${ubicacion.lon}. **DEBES usar ESTA ubicación para todas las consultas de clima y eventos locales.** Ignora cualquier otra ubicación que puedas inferir de la IP.`;
        const existe = historial.some(msg => 
            msg.role === 'system' && msg.content.includes('Ubicación del usuario')
        );
        if (existe) {
            const index = historial.findIndex(msg => 
                msg.role === 'system' && msg.content.includes('Ubicación del usuario')
            );
            if (index !== -1) {
                historial[index].content = mensajeContexto;
                guardarHistorial(historial);
            }
        } else {
            historial.unshift({ role: 'system', content: mensajeContexto });
            guardarHistorial(historial);
        }
    }

    function crearBurbuja() {
        const existing = document.getElementById('nomi-bubble');
        if (existing) existing.remove();

        const bubble = document.createElement('div');
        bubble.id = 'nomi-bubble';
        bubble.textContent = '💬';
        bubble.style.cssText = `
            position: fixed;
            left: 20px;
            bottom: 20px;
            width: 56px;
            height: 56px;
            border-radius: 50%;
            background: #1a1a2e;
            border: 2px solid #FF6B6B;
            color: white;
            font-size: 28px;
            cursor: pointer;
            z-index: 999998;
            box-shadow: 0 4px 16px rgba(255,107,107,0.5);
            display: flex;
            align-items: center;
            justify-content: center;
            user-select: none;
            touch-action: none;
            transition: transform 0.2s;
        `;

        if (posicionBurbuja.x || posicionBurbuja.y) {
            bubble.style.left = posicionBurbuja.x + 'px';
            bubble.style.top = posicionBurbuja.y + 'px';
            bubble.style.right = 'auto';
            bubble.style.bottom = 'auto';
        }

        document.body.appendChild(bubble);
        burbujaVisible = true;

        let startX, startY, origX, origY, isDragging = false, isTouching = false;

        bubble.addEventListener('touchstart', (e) => {
            if (ventanaAbierta) return;
            const touch = e.touches[0];
            startX = touch.clientX;
            startY = touch.clientY;
            origX = bubble.offsetLeft || 0;
            origY = bubble.offsetTop || 0;
            isDragging = false;
            isTouching = true;
            bubble.style.cursor = 'grabbing';
        }, { passive: true });

        bubble.addEventListener('touchmove', (e) => {
            if (!isTouching || ventanaAbierta) return;
            const touch = e.touches[0];
            const dx = touch.clientX - startX;
            const dy = touch.clientY - startY;
            if (Math.sqrt(dx*dx + dy*dy) > 10) {
                isDragging = true;
                let newX = origX + dx;
                let newY = origY + dy;
                newX = Math.max(0, Math.min(window.innerWidth - 56, newX));
                newY = Math.max(0, Math.min(window.innerHeight - 56, newY));
                bubble.style.left = newX + 'px';
                bubble.style.top = newY + 'px';
                bubble.style.right = 'auto';
                bubble.style.bottom = 'auto';
                posicionBurbuja = { x: newX, y: newY };
                setPosicion(posicionBurbuja);
            }
        }, { passive: true });

        bubble.addEventListener('touchend', () => {
            isTouching = false;
            bubble.style.cursor = 'pointer';
            if (!isDragging) {
                toggleVentana();
            }
            isDragging = false;
        }, { passive: true });

        let mouseDown = false, mouseStartX, mouseStartY, mouseOrigX, mouseOrigY, mouseDragging = false;
        bubble.addEventListener('mousedown', (e) => {
            if (ventanaAbierta) return;
            mouseDown = true;
            mouseStartX = e.clientX;
            mouseStartY = e.clientY;
            mouseOrigX = bubble.offsetLeft || 0;
            mouseOrigY = bubble.offsetTop || 0;
            mouseDragging = false;
            bubble.style.cursor = 'grabbing';
        });

        document.addEventListener('mousemove', (e) => {
            if (!mouseDown || ventanaAbierta) return;
            const dx = e.clientX - mouseStartX;
            const dy = e.clientY - mouseStartY;
            if (Math.sqrt(dx*dx + dy*dy) > 5) {
                mouseDragging = true;
                let newX = mouseOrigX + dx;
                let newY = mouseOrigY + dy;
                newX = Math.max(0, Math.min(window.innerWidth - 56, newX));
                newY = Math.max(0, Math.min(window.innerHeight - 56, newY));
                bubble.style.left = newX + 'px';
                bubble.style.top = newY + 'px';
                bubble.style.right = 'auto';
                bubble.style.bottom = 'auto';
                posicionBurbuja = { x: newX, y: newY };
                setPosicion(posicionBurbuja);
            }
        });

        document.addEventListener('mouseup', () => {
            if (mouseDown) {
                mouseDown = false;
                bubble.style.cursor = 'pointer';
                if (!mouseDragging) {
                    toggleVentana();
                }
                mouseDragging = false;
            }
        });
    }

    // ======== PREGUNTAR A LA IA ========
    async function preguntar(texto) {
        const input = document.getElementById('nomi-input');
        if (input) input.value = '';

        if (!texto.trim() || isWaiting) return;

        if (texto.trim() === '!cmd' || texto.trim() === '!comandos') {
            mostrarAyuda();
            return;
        }

        isWaiting = true;
        document.getElementById('nomi-modelo-display').textContent = '⏳ pensando...';

        if (ubicacionActivada && ubicacionActual) {
            const ahora = Date.now();
            if (ahora - ubicacionActual.timestamp > UBICACION_EXPIRACION) {
                actualizarUbicacion(true);
            }
        }

        // ======== DETECTAR COMANDO DE BÚSQUEDA ========
        const cmdBusqueda = texto.match(/^(investiga|busca|investigar|buscar)\s*[:|]?\s*(.+)/i);
        let esBusqueda = false;
        let consulta = '';
        if (cmdBusqueda) {
            consulta = cmdBusqueda[2].trim();
            if (consulta) {
                esBusqueda = true;
                if (!busquedaWebActiva) {
                    busquedaWebTemporal = true;
                }
            }
        }

        if (busquedaForzada) {
            esBusqueda = true;
            consulta = texto.trim();
            busquedaForzada = false;
            if (!busquedaWebActiva) {
                busquedaWebTemporal = true;
            }
        }

        if (!esBusqueda && busquedaWebActiva && requiereBusqueda(texto)) {
            esBusqueda = true;
            consulta = texto.trim();
            console.log('🔍 Búsqueda automática activada por detección de palabras clave.');
        }

        if (esBusqueda && consulta) {
            ocultarCargando();
            document.getElementById('nomi-modelo-display').textContent = modeloActual;
            await procesarBusqueda(consulta);
            if (busquedaWebTemporal) {
                busquedaWebTemporal = false;
                console.log('🔍 Búsqueda web temporal desactivada.');
            }
            isWaiting = false;
            return;
        }

        // ======== ANÁLISIS DE PÁGINA (MODO WEB) ========
        const palabrasClave = ['analiza', 'examina', 'escanea', 'resume esta página'];
        const esAnalisis = palabrasClave.some(p => texto.toLowerCase().includes(p));
        const usarWeb = modoWebActivo || esAnalisis;
        let infoPagina = '';
        if (usarWeb) {
            const limite = modoLigeroActivo ? LIMITE_TEXTO_LIGERO : LIMITE_TEXTO_NORMAL;
            const info = extraerInformacionPagina(limite);
            infoPagina = `
    INFORMACIÓN DE LA PÁGINA ACTUAL:
    Título: ${info.titulo}
    URL: ${info.url}
    Descripción: ${info.metaDesc}
    Encabezados: ${info.encabezados.join(', ')}
    Contenido principal (primeros ${limite} caracteres):
    ${info.texto}
    ---
    `;
            if (modoWebActivo) {
                modoWebActivo = false;
                document.getElementById('nomi-web-status').style.display = 'none';
                document.getElementById('nomi-web-btn').style.borderColor = '#555';
                document.getElementById('nomi-web-btn').style.color = '#888';
            }
        }

        // ======== CONSTRUIR CONTEXTO ========
        const contextoFechaHora = obtenerContextoTiempo();
        let contextoUbicacion = '';
        if (ubicacionActivada && ubicacionActual && requiereUbicacion(texto)) {
            const fuente = ubicacionActual.fuente === 'gps' ? 'GPS' : 'IP aproximada';
            contextoUbicacion = `📍 Ubicación del usuario: ${ubicacionActual.ciudad}, ${ubicacionActual.pais} (${fuente}). Coordenadas GPS: ${ubicacionActual.lat}, ${ubicacionActual.lon}. **DEBES usar ESTA ubicación.**`;
        } else if (ubicacionActivada && !ubicacionActual && requiereUbicacion(texto)) {
            contextoUbicacion = '📍 Ubicación: no disponible (solicitando...)';
        }

        const contextoCompleto = contextoUbicacion ? `${contextoFechaHora}\n${contextoUbicacion}` : contextoFechaHora;

        // ======== CONSTRUIR MENSAJES PARA LA IA ========
        let mensajesParaEnviar = [];
        if (modoResumenActivo && contextoSeleccionado === 10 && resumenPersistente) {
            mensajesParaEnviar.push({
                role: 'system',
                content: `Resumen de la conversación anterior:\n${resumenPersistente}`
            });
        }
        const limiteMensajes = Math.min(contextoSeleccionado, CONTEXTO_RECIENTE);
        const mensajesRecientes = historial
            .filter(m => m.role === 'user' || m.role === 'assistant')
            .slice(-limiteMensajes);
        mensajesParaEnviar.push(...mensajesRecientes);

        let mensajeCompleto = texto;
        if (infoPagina) {
            mensajeCompleto = infoPagina + '\nPregunta del usuario: ' + texto;
        }

        const mensajeFinal = `${contextoCompleto}\n\nPregunta del usuario: ${mensajeCompleto}`;

        historial.push({role: 'user', content: texto});
        guardarHistorial(historial);
        agregarMensaje('yo', texto);

        const enviar = document.getElementById('nomi-enviar');
        if (input) input.disabled = true;
        if (enviar) enviar.disabled = true;
        mostrarCargando();

        try {
            const respuesta = await fetch(urlBaseActual + '/chat/completions', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': 'Bearer ' + apiKeyActual,
                    'HTTP-Referer': window.location.href,
                    'X-Title': 'NoMi Asistente'
                },
                body: JSON.stringify({
                    model: modeloActual,
                    messages: [
                        ...mensajesParaEnviar,
                        {role: 'user', content: mensajeFinal}
                    ],
                    stream: false
                })
            });
            ocultarCargando();
            const datos = await respuesta.json();
            if (datos.choices && datos.choices[0]) {
                const respuestaTexto = datos.choices[0].message.content;
                document.getElementById('nomi-modelo-display').textContent = modeloActual;
                if (datos.usage) {
                    tokens.total += datos.usage.total_tokens || 0;
                    tokens.input += datos.usage.prompt_tokens || 0;
                    tokens.output += datos.usage.completion_tokens || 0;
                    setTokens(tokens);
                }
                contadorPreguntas++;
                setContador(contadorPreguntas);
                historial.push({role: 'assistant', content: respuestaTexto});
                guardarHistorial(historial);
                agregarMensaje('bot', respuestaTexto);
                actualizarStats();
                if (modoResumenActivo && contextoSeleccionado === 10) {
                    setTimeout(() => generarResumen(historial), 100);
                }
            } else if (datos.error) {
                agregarMensaje('bot', '❌ Error: ' + (datos.error.message || JSON.stringify(datos.error)));
                historial.pop();
                guardarHistorial(historial);
                document.getElementById('nomi-modelo-display').textContent = '⚠️ error';
                registrarError('api', datos.error.message || 'Error desconocido en API', `Modelo: ${modeloActual}`);
            } else {
                agregarMensaje('bot', '❌ Error inesperado');
                historial.pop();
                guardarHistorial(historial);
                document.getElementById('nomi-modelo-display').textContent = '⚠️ error';
                registrarError('script', 'Respuesta inesperada de la API', 'Sin detalles');
            }
        } catch (error) {
            ocultarCargando();
            agregarMensaje('bot', '❌ Error de conexión: ' + error.message);
            historial.pop();
            guardarHistorial(historial);
            document.getElementById('nomi-modelo-display').textContent = '⚠️ error';
            registrarError('network', error.message, `URL: ${urlBaseActual}`);
        }
        if (input) input.disabled = false;
        if (enviar) enviar.disabled = false;
        if (input) input.focus();
        isWaiting = false;
    }

                mostrarNotificacionTemporal(`✅ Contexto actualizado a ${contextoSeleccionado} mensajes.`);

        };

                return;

    function exportarChat(fecha, formato) {
        const key = 'nomi_historial_' + getPageKey() + '_' + fecha;
        const data = localStorage.getItem(key);
        if (!data) { alert('No hay historial para esa fecha.'); return; }
        const historialData = JSON.parse(data);
        let contenido = '';
        if (formato === 'json') {
            contenido = JSON.stringify(historialData, null, 2);
        } else {
            contenido = historialData.map(m => {
                const rol = m.role === 'user' ? '👤 Tú' : `🤖 ${NOMBRE_ASISTENTE}`;
                return `${rol}: ${m.content}`;
            }).join('\n\n');
        }
        const blob = new Blob([contenido], {type: formato === 'json' ? 'application/json' : 'text/plain;charset=utf-8'});
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `chat_${fecha}.${formato}`;
        a.click();
        URL.revokeObjectURL(url);
    }

    function agregarMensaje(quien, texto) {
        const chatBody = document.getElementById('nomi-chat-body');
        if (!chatBody) return;
        const color = quien === 'yo' ? '#FF6B6B' : '#34a853';
        const nombre = quien === 'yo' ? 'Tú' : NOMBRE_ASISTENTE;

    function cargarHistorial() {
        const chatBody = document.getElementById('nomi-chat-body');
        if (!chatBody) return;
        chatBody.innerHTML = '';
        const mensajesMostrar = historial
            .filter(msg => msg.role !== 'system')
            .slice(-MENSAJES_VISIBLES);
        if (mensajesMostrar.length === 0) {
            const empty = document.createElement('div');
            empty.style.cssText = 'color:#666; text-align:center; padding:20px 0;';
            empty.textContent = '💬 Sin mensajes aún. ¡Pregunta algo!';
            chatBody.appendChild(empty);
            return;
        }
        mensajesMostrar.forEach(msg => {
            const esUsuario = msg.role === 'user';
            const color = esUsuario ? '#FF6B6B' : '#34a853';
            const nombre = esUsuario ? 'Tú' : NOMBRE_ASISTENTE;
            const msgDiv = document.createElement('div');
            msgDiv.style.cssText = `
                margin:4px 0; padding:6px 10px; border-radius:10px;
                background:${color}33; border-left:3px solid ${color};
                font-size:12px; word-wrap:break-word;
            `;
            msgDiv.innerHTML = `<b style="color:${color};">${nombre}:</b> ${msg.content}`;
            chatBody.appendChild(msgDiv);
        });
        chatBody.scrollTop = chatBody.scrollHeight;
    }

    function extraerInformacionPagina(limite) {
        const titulo = document.title || 'Sin título';
        const url = window.location.href;
        const metaDesc = document.querySelector('meta[name="description"]')?.content || 'Sin descripción';
        const main = document.querySelector('main') || document.querySelector('article');
        const cuerpo = (main?.innerText || document.body.innerText || document.body.textContent || '')
            .replace(/\s+/g, ' ').replace(/\n/g, ' ').trim().slice(0, limite);
        const encabezados = [];
        document.querySelectorAll('h1, h2, h3').forEach(h => encabezados.push(h.textContent.trim()));
        return { titulo, url, metaDesc, texto: cuerpo, encabezados: encabezados.slice(0,10) };
    }

    function mostrarCargando() {
        const loading = document.getElementById('nomi-loading');
        if (loading) {
            loading.style.display = 'block';
            let count = 0;
            const dots = document.getElementById('nomi-dots');
            if (dots) {
                const interval = setInterval(() => {
                    count = (count % 3) + 1;
                    dots.textContent = '.'.repeat(count);
                }, 400);
                loading.dataset.interval = interval;
            }
        }
    }
    function ocultarCargando() {
        const loading = document.getElementById('nomi-loading');
        if (loading) {
            loading.style.display = 'none';
            if (loading.dataset.interval) {
                clearInterval(parseInt(loading.dataset.interval));
                delete loading.dataset.interval;
            }
        }
    }

    async function generarResumen(historialCompleto) {
        if (!modoResumenActivo || contextoSeleccionado !== 10 || historialCompleto.length < 4) return;
        try {
            const respuesta = await fetch(urlBaseActual + '/chat/completions', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': 'Bearer ' + apiKeyActual
                },
                body: JSON.stringify({
                    model: modeloActual,
                    messages: [
                        {
                            role: 'system',
                            content: 'Eres un asistente que resume conversaciones. Genera un resumen COMPACTO (máximo 300 palabras) de toda la conversación. Incluye temas principales y decisiones. Responde SOLO con el resumen.'
                        },
                        {
                            role: 'user',
                            content: `Resume esta conversación:\n\n${historialCompleto.map(m => `${m.role === 'user' ? 'Usuario' : 'Asistente'}: ${m.content}`).join('\n')}`
                        }
                    ],
                    stream: false,
                    max_tokens: 500
                })
            });
            const datos = await respuesta.json();
            if (datos.choices && datos.choices[0]) {
                setResumen(datos.choices[0].message.content);
                console.log('🧠 Resumen actualizado.');
            }
        } catch (error) {
            registrarError('api', error.message, 'Generación de resumen');
            console.log('⚠️ Error generando resumen:', error.message);
        }
    }
